287  static sl::io::span<char> make_header(std::array<char, 10>& buf, frame_type fr_type, size_t pl_len,
288  bool masked = false, bool partial = false) {
289  uint8_t mask_byte = !masked ? 0 : (1<<7);
290  buf[0] = (!partial ? (1<<7) : 0) | static_cast<uint8_t>(fr_type);
291  if (pl_len < (1<<7) - 2) {
292  buf[1] = mask_byte | pl_len;
293  return sl::io::make_span(buf.data(), 2);
294  } else {
295  auto sink = sl::io::memory_sink({buf.data() + 2, buf.size() - 2});
296  if(pl_len < (1<<16)) {
297  auto val = mask_byte | ((1<<7) - 2);
298  buf[1] = static_cast<uint8_t>(val);
299  sl::endian::write_16_be(sink, pl_len);
300  return sl::io::make_span(buf.data(), 4);
301  } else {
302  auto val = mask_byte | ((1<<7) - 1);
303  buf[1] = static_cast<uint8_t>(val);
304  sl::endian::write_64_be(sink, pl_len);
305  return sl::io::make_span(buf.data(), buf.size());
306  }
307  }

358  void parse_payload_64() {
359  if (parsing && 127 == payload_len_7) {
360  if (view.size() >= 10) {
361  auto src = sl::io::array_source(view.data() + prefix_len, 8);
362  auto p64 = sl::endian::read_64_be<uint64_t>(src);
363  if (p64 < std::numeric_limits<int32_t>::max() -
364  (prefix_len + 8 + sizeof(mask))) {
365  this->payload_len = static_cast<uint32_t>(p64);
366  this->ex_payload_len_field_size = 8;

372  this->parsing = false;
373  }

377  void parse_mask() {
378  if (parsing) {
379  this->masked = 1 == ((view[1] >> 7) & 0x01);
380  if (masked && view.size() >= prefix_len + ex_payload_len_field_size + 4) {
381  auto src = sl::io::array_source(view.data() + prefix_len + ex_payload_len_field_size, 4);
382  this->mask = sl::endian::read_32_be<uint32_t>(src);
383  if (mask > 0) {
